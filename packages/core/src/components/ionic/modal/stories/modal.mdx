## Usage

```ts
// modal-example.ts

async function openModal() {
  const modal = await modalController.create({
    component: 'modal-example',
    cssClass: 'tp-modal'
  });

  await modal.present();
}

async function closeModal() {
  await modal.dismiss();
}
```
